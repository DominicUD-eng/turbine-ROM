class Node:
    """
    Holds geometric + state data at a radial location r_i.
    State vector component is v_theta[i]. v_r is algebraic (from continuity).
    """
    def __init__(self, r: float):
        self.r = r
        self.vtheta = 0.0   # tangetial velocity at this node [m/s]
        self.vr = 0.0       # radial velocity [m/s]
